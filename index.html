Q1 Problem: You are given a string (STR) of length N, consisting of only the lower case English alphabet. Your task is to remove all the duplicate occurrences of characters in the string.

Input: abcadeecfb

Output: abcdef

function removeDuplicates(inputString) {
            
            const encounteredChars = {};
          
            
            let outputString = '';
          
            for (const char of inputString) {
              
              if (!encounteredChars[char]) {

                encounteredChars[char] = true;
              }
            }
          
            return outputString;
          }

          const inputString = 'abcadeecfb';
          const outputString = removeDuplicates(inputString);
          console.log(outputString);

Output: abcdef


Q2 Problem: You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps)

Input: abcadeecfb

Output: a=2 b=2 c=2 d=1 e=2 f=1

function countAlphabets(inputString) {
                
                const charCountMap = new Map();
            
                const lowercaseInput = inputString.toLowerCase();
              
                for (const char of lowercaseInput) {
                  
                  if (/[a-z]/.test(char)) {
                    
                    if (charCountMap.has(char)) {

                      charCountMap.set(char, charCountMap.get(char) + 1);
                    }
                     else {

                      charCountMap.set(char, 1);
                    }
                  }
                }
              
                charCountMap.forEach((count, char) => {

                  console.log(`${char}: ${count}`);

                });
              }
              
              const inputString = "abcadeecfb";

              countAlphabets(inputString);

Output a: 2 b: 2 c: 2 d: 1 e: 3 f: 1
